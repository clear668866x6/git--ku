=== 4.2 综合测试用例2：递归函数 + 嵌套条件 ===
int factor(int m);
int main(void) {
    int n, result;
    n = read();
    if (n <= 0) {
        result = 0;
    } else {
        if (n > 8) {
            result = -1;
        } else {
            result = factor(n);
        }
    }
    write(result);
    return result;
}
int factor(int m) {
    int fa;
    if (m <= 1) {
        fa = 1;
    } else {
        fa = m * factor(m - 1);
    }
    return fa;
}
